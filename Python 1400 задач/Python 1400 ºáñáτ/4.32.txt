num = int(input("Введите трехзначное число: "))
a = num // 100 # первая цифра
b = (num // 10) % 10 # вторая цифра
c = num % 10 # третья цифра

if num ** 2 == a ** 3 + b ** 3 + c ** 3:
    print("Квадрат числа", num, "равен сумме кубов его цифр")
else:
    print("Квадрат числа", num, "не равен сумме кубов его цифр")
