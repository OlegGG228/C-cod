#Первый вариант:

grades = [] # создаем пустой список для оценок

for i in range(12): # цикл по количеству учеников
  print("Введите оценки для ученика №", i+1) # выводим номер ученика
  row = input().split() # вводим оценки для ученика и разделяем их
  row = [int(x) for x in row] # преобразуем строки в числа
  grades.append(row) # добавляем оценки в список

summa = 0 # переменная для хранения суммы оценок

for i in range(12): # цикл по количеству учеников
  for j in range(3): # цикл по количеству предметов
    summa += grades[i][j] # добавляем оценку к сумме

print("Сумма оценок:", summa) # выводим сумму оценок

#Второй вариант:

grades = [[0 for j in range(3)] for i in range(12)] # создаем список для оценок

for i in range(3): # цикл по количеству предметов
  print("Введите оценки для предмета №", i+1) # выводим номер предмета
  column = input().split() # вводим оценки для предмета и разделяем их
  column = [int(x) for x in column] # преобразуем строки в числа

  for j in range(12): # цикл по количеству учеников
    grades[j][i] = column[j] # добавляем оценку в таблицу

summa = 0 # переменная для хранения суммы оценок

for i in range(12): # цикл по количеству учеников
  for j in range(3): # цикл по количеству предметов
    summa += grades[i][j] # добавляем оценку к сумме

print("Сумма оценок:", summa) # выводим сумму оценок
