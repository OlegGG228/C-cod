grades = [4, 5, 3, 5, 2, 4, 5, 5, 4, 4, 5, 4] # оценки ученика
has_triple = False # флаг для обнаружения троек

i = 0 # начальный индекс
while i < len(grades): # пока не проверили все оценки
    if grades[i] == 3: # если обнаружена тройка
        has_triple = True # изменить флаг на True
        break # выйти из цикла
    i += 1 # увеличить индекс для проверки следующей оценки

if has_triple: # если была обнаружена тройка
    print("Ученик имеет тройку.") # сообщение об ошибке
else: # если троек нет
    print("Ученик не имеет троек.") # сообщение об успехе
