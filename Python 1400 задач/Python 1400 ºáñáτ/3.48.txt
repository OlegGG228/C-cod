угловые_градусы = float(input("Введите количество градусов, на которое повернулась часовая стрелка: "))

# Определяем количество полных часов
полные_часы = int(угловые_градусы // 30)

# Определяем количество полных минут 
полные_минуты = int((угловые_градусы % 30) * 2)

print("С начала суток прошло", полные_часы, "часов и", полные_минуты, "минут.")
