population = [15, 17, 12, 20, 8, 14, 9, 11, 16, 19, 13, 7] # количество жителей в каждом районе в тысячах человек
area = [20, 30, 35, 22, 15, 28, 18, 21, 25, 27, 31, 16] # площадь каждого района в км2

total_population = sum(population) # общее количество жителей в области
total_area = sum(area) # общая площадь области

average_density = total_population / total_area # средняя плотность населения по области в целом

print("Средняя плотность населения по области в целом: %.2f тыс. чел./км2" % average_density)
