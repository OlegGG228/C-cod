#Первый вариант программы:

a = float(input("Введите число а: "))   # ввод числа а
n = 2   # начальное значение n
sum = 1 + 1/n   # вычисление первого числа из последовательности
while sum >= a:   # пока последнее вычисленное число из последовательности не меньше а
    print(sum)   # печать числа
    n += 1   # переход к следующему значению n
    sum = 1 + 1/n   # вычисление следующего числа из последовательности

#Второй вариант программы:

a = float(input("Введите число а: "))   # ввод числа а
n = 2   # начальное значение n
while True:   # бесконечный цикл
    sum = 1 + 1/n   # вычисление числа из последовательности
    if sum < a:   # если вычисленное число меньше а
        break   # выход из цикла
    print(sum)   # печать числа
    n += 1   # переход к следующему значению n

#Третий вариант программы:

a = float(input("Введите число а: "))   # ввод числа а
n = 2   # начальное значение n
seq = []   # инициализация списка для хранения чисел из последовательности
while True:   # бесконечный цикл
    sum = 1 + 1/n   # вычисление числа из последовательности
    if sum < a:   # если вычисленное число меньше а
        break   # выход из цикла
    seq.append(sum)   # добавление числа в список
    n += 1   # переход к следующему значению n
for s in seq:   # цикл по числам из списка
    print(s)   # печать числа
