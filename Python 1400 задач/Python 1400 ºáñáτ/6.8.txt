#Первый вариант программы:

n = 1
p = int(input("Введите значение для определения первого натурального числа, квадрат которого больше этого значения: "))
while n <= 100:
    if n**2 > p:
        print("Первое натуральное число, квадрат которого больше ",p," - ",n)
        break
    n += 1
    
#Второй вариант программы:

p = int(input("Введите значение для определения первого натурального числа, квадрат которого больше этого значения: "))
for n in range(1,101):
    if n**2 > p:
        print("Первое натуральное число, квадрат которого больше ",p," - ",n)
        break
