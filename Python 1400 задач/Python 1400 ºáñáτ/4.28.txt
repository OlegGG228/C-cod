number = int(input("Введите двузначное число: "))
sum = (number // 10) + (number % 10) # находим сумму цифр
if sum >= 10: # проверяем, двузначное число или нет
    print("Сумма цифр является двузначным числом")
else:
    print("Сумма цифр не является двузначным числом")
if number > sum:
    print("Число больше суммы своих цифр")
else:
    print("Число не больше суммы своих цифр")
