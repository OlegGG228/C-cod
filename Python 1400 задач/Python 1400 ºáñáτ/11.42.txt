arr = [5, 8, 2, 9, 3, 10, 7]
k1 = 2
k2 = 5
s1 = 1
s2 = 6

# сумма всех элементов массива
sum_all = sum(arr)
print("Сумма всех элементов массива: ", sum_all)

# произведение всех элементов массива
prod_all = 1
for i in arr:
    prod_all *= i
print("Произведение всех элементов массива: ", prod_all)

# сумма квадратов всех элементов массива
sum_squares = sum([i**2 for i in arr])
print("Сумма квадратов всех элементов массива: ", sum_squares)

# сумма первых шести элементов массива
sum_first_six = sum(arr[:6])
print("Сумма первых шести элементов массива: ", sum_first_six)

# сумма элементов массива с k1-го по k2-й
sum_k1_k2 = sum(arr[k1:k2+1])
print(f"Сумма элементов массива с {k1}-го по {k2}-й: ", sum_k1_k2)

# среднее арифметическое всех элементов массива
avg_all = sum_all / len(arr)
print("Среднее арифметическое всех элементов массива: ", avg_all)

# среднее арифметическое элементов массива с s1-го по s2-й
avg_s1_s2 = sum(arr[s1-1:s2]) / (s2 - s1 + 1)
print(f"Среднее арифметическое элементов массива с {s1}-го по {s2}-й: ", avg_s1_s2)
