num = int(input("Введите натуральное число: "))

# Перебираем числа от 2 до num и проверяем, является ли num на них делителем
for i in range(2, num+1):
    if num % i == 0:
        print("Наименьший делитель числа", num, "равен", i)
        break
else:
    print("Введенное число не имеет делителей, отличных от 1")
