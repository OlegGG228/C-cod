# инициализация переменной для суммы нечетных чисел
summa = 0

# цикл от 1 до 49 с шагом 2 (чтобы получить нечетные числа)
for i in range(1, 50, 2):
    # проверяем, что число меньше 50 и положительное
    if i < 50 and i > 0:
        # добавляем число к сумме
        summa += i

# выводим результат на экран
print("Сумма положительных нечетных чисел, меньших 50, равна:", summa)
