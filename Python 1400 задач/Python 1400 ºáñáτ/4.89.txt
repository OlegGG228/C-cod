a = int(input("Введите часы прибытия поезда: "))
b = int(input("Введите минуты прибытия поезда: "))
c = int(input("Введите часы отправки поезда: "))
d = int(input("Введите минуты отправки поезда: "))
p = int(input("Введите часы прихода пассажира на платформу: "))
t = int(input("Введите минуты прихода пассажира на платформу: "))

if a < c:
    if p < a or (p == a and t <= b):
        print("Поезд стоит на платформе")
    elif p > c or (p == c and t >= d):
        print("Поезд уже ушел")
    else:
        print("Поезд прибыл, но уже ушел")
elif a == c:
    if b <= t < d:
        print("Поезд стоит на платформе")
    elif t < b:
        print("Поезд прибыл, но еще не пришел")
    else:
        print("Поезд прибыл, но уже ушел")
else:
    if p > c or (p == c and t >= d):
        print("Поезд уже ушел")
    elif p < a or (p == a and t <= b):
        print("Поезд стоит на платформе")
    else:
        print("Поезд прибудет позже")
