
# объявляем исходный массив и значение s
sales = [100, 200, 150, 300, 250, 400, 100, 150, 200, 250, 300, 350]
s = 500

# переменная для подсчета количества дней, в которые стоимость продаж была больше s
count = 0

# проходим циклом по массиву и проверяем каждый элемент
for sale in sales:
    if sale > s:
        count += 1

# выводим результат
print("Количество дней, в которые стоимость проданных товаров превысила", s, ":", count)
