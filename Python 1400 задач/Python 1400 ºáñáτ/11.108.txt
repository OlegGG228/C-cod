arr = [10, -4, 12, 56, -4, -89] # заданный массив
count = 0 # счетчик количества смен знака
prev = arr[0] # предыдущий элемент
for i in range(1, len(arr)): # перебираем элементы массива, начиная со второго
    if prev * arr[i] < 0: # если знаки предыдущего и текущего элементов различны
        count += 1 # увеличиваем счетчик на 1
    prev = arr[i] # сохраняем текущий элемент как предыдущий для следующей итерации
print(count) # выводим количество смен знака
