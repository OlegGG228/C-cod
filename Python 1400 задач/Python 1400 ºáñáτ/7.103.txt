# исходные данные
teams = {
    'Команда A': 20,
    'Команда B': 30,
    'Команда C': 25,
    'Команда D': 15,
    'Команда E': 10
}

# сортируем словарь по значениям в порядке убывания
sorted_teams = sorted(teams.items(), key=lambda x: x[1], reverse=True)

# получаем список команд в порядке убывания набранных очков
sorted_teams_keys = [team[0] for team in sorted_teams]

# сравниваем списки и выводим результат
if list(teams.keys()) == sorted_teams_keys:
    print('Команды перечислены в списке в соответствии с занятыми ими местами в чемпионате')
else:
    print('Команды не перечислены в списке в соответствии с занятыми ими местами в чемпионате')
