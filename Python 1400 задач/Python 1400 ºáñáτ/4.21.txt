left1_x = int(input("Введите координату X левого нижнего угла первого прямоугольника: "))
left1_y = int(input("Введите координату Y левого нижнего угла первого прямоугольника: "))
width1 = int(input("Введите ширину первого прямоугольника: "))
height1 = int(input("Введите высоту первого прямоугольника: "))

left2_x = int(input("Введите координату X левого нижнего угла второго прямоугольника: "))
left2_y = int(input("Введите координату Y левого нижнего угла второго прямоугольника: "))
width2 = int(input("Введите ширину второго прямоугольника: "))
height2 = int(input("Введите высоту второго прямоугольника: "))

min_x = min(left1_x, left2_x)
min_y = min(left1_y, left2_y)
max_x = max(left1_x + width1, left2_x + width2)
max_y = max(left1_y + height1, left2_y + height2)

print("Координаты левого нижнего угла минимального прямоугольника: ({0}, {1})".format(min_x, min_y))
print("Координаты правого верхнего угла минимального прямоугольника: ({0}, {1})".format(max_x, max_y))
