a = int(input("Введите длину прямоугольного параллелепипеда: "))
b = int(input("Введите ширину прямоугольного параллелепипеда: "))
c = int(input("Введите высоту прямоугольного параллелепипеда: "))

volume = a * b * c
lateral_area = 2 * (a + b) * c

print("Объем прямоугольного параллелепипеда:", volume)
print("Площадь боковой поверхности прямоугольного параллелепипеда:", lateral_area)
