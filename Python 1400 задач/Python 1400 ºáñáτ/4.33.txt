
num = int(input("Введите трехзначное число: "))
a = int(input("Введите число a: "))

# вычисление цифр числа
a1 = num // 100
a2 = (num % 100) // 10
a3 = num % 10

# а) является ли сумма его цифр двузначным числом
summa = a1 + a2 + a3
if summa >= 10 and summa <= 99:
    print("Сумма цифр", num, "является двузначным числом")
else:
    print("Сумма цифр", num, "не является двузначным числом")

# б) является ли произведение его цифр трехзначным числом
proizv = a1 * a2 * a3
if proizv >= 100 and proizv <= 999:
    print("Произведение цифр", num, "является трехзначным числом")
else:
    print("Произведение цифр", num, "не является трехзначным числом")

# в) больше ли числа а произведение его цифр
if proizv > a:
    print("Произведение цифр", num, "больше числа", a)
else:
    print("Произведение цифр", num, "не больше числа", a)

# г) кратна ли пяти сумма его цифр
if summa % 5 == 0:
    print("Сумма цифр", num, "кратна пяти")
else:
    print("Сумма цифр", num, "не кратна пяти")

# д) кратна ли сумма его цифр числу а
if summa % a == 0:
    print("Сумма цифр", num, "кратна числу", a)
else:
    print("Сумма цифр", num, "не кратна числу", a)
