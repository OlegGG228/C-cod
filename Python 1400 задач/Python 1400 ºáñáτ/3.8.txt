# создаем переменную для серии билета
series = "AB"
# создаем переменную для номера первого билета
number = 1643
# создаем цикл для каждого ряда мест
for row in range(1, 21):
    # создаем цикл для каждого кресла в ряду
    for seat in range(1, 16):
        # выводим информацию о билете
        print("Серия:", series, " Номер:", number, " Ряд:", row, " Место:", seat)
        # увеличиваем номер на 1 для следующего билета
        number += 1
