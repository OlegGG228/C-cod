number = 1234

# 1) число, полученное при прочтении его цифр справа налево
rev_number = int(str(number)[::-1])
print("Число, полученное при прочтении его цифр справа налево:", rev_number)

# 2) число, образуемое при перестановке первой и второй, третьей и четвертой цифр заданного числа
swap_12_34_number = int(str(number)[1] + str(number)[0] + str(number)[3] + str(number)[2])
print("Число, образуемое при перестановке первой и второй, третьей и четвертой цифр заданного числа:", swap_12_34_number)

# 3) число, образуемое при перестановке второй и третьей цифр заданного числа
swap_23_number = int(str(number)[0] + str(number)[2] + str(number)[1] + str(number)[3])
print("Число, образуемое при перестановке второй и третьей цифр заданного числа:", swap_23_number)

# 4) число, образуемое при перестановке двух первых и двух последних цифр заданного числа
swap_first_last_number1 = int(str(number)[2:] + str(number)[:2])
print("Число, образуемое при перестановке двух первых и двух последних цифр заданного числа (способ 1):", swap_first_last_number1)

swap_first_last_number2 = (number % 100) * 100 + (number // 100)
print("Число, образуемое при перестановке двух первых и двух последних цифр заданного числа (способ 2):", swap_first_last_number2)
