f = int(input("Введите первый член арифметической прогрессии: "))
s = int(input("Введите шаг арифметической прогрессии: "))
p = int(input("Введите число, которое нужно проверить: "))

if (p - f) % s == 0:
    print(p, "является членом арифметической прогрессии")
else:
    print(p, "не является членом арифметической прогрессии")
