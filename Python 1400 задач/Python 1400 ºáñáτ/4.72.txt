A_x = int(input("Введите координату x левого нижнего угла прямоугольника А: "))
A_y = int(input("Введите координату y левого нижнего угла прямоугольника А: "))
A_width = int(input("Введите ширину прямоугольника А: "))
A_height = int(input("Введите высоту прямоугольника А: "))

# Прямоугольник В (координаты левого нижнего угла, ширина, высота)
B_x = int(input("Введите координату x левого нижнего угла прямоугольника В: "))
B_y = int(input("Введите координату y левого нижнего угла прямоугольника В: "))
B_width = int(input("Введите ширину прямоугольника В: "))
B_height = int(input("Введите высоту прямоугольника В: "))

# Определяем координаты правого верхнего угла прямоугольника А
A_right_x = A_x + A_width
A_right_y = A_y + A_height

# Определяем координаты правого верхнего угла прямоугольника В
B_right_x = B_x + B_width
B_right_y = B_y + B_height

# Проверяем принадлежность всех точек первого прямоугольника второму

if (B_x <= A_x <= B_right_x or B_x <= A_right_x <= B_right_x) and (B_y <= A_y <= B_right_y or B_y <= A_right_y <= B_right_y):
    print("Все точки прямоугольника А принадлежат прямоугольнику В")
else:
    print("Не все точки прямоугольника А принадлежат прямоугольнику В")

# Проверяем принадлежность всех точек одного прямоугольника другому

if (A_x <= B_x <= A_right_x or A_x <= B_right_x <= A_right_x) and (A_y <= B_y <= A_right_y or A_y <= B_right_y <= A_right_y):
    print("Все точки прямоугольника В принадлежат прямоугольнику А")
else:
    print("Не все точки прямоугольника В принадлежат прямоугольнику А")

# Проверяем пересечение прямоугольников

if (A_x <= B_x <= A_right_x or A_x <= B_right_x <= A_right_x or B_x <= A_x <= B_right_x or B_x <= A_right_x <= B_right_x) and (A_y <= B_y <= A_right_y or A_y <= B_right_y <= A_right_y or B_y <= A_y <= B_right_y or B_y <= A_right_y <= A_right_y):
    print("Прямоугольники пересекаются")
else:
    print("Прямоугольники не пересекаются")
