heights = []  # создаем пустой список для хранения ростов
count = 0  # создаем счетчик для количества юношей с ростом менее 165 см

# заполняем список ростами
for i in range(12):
    height = int(input("Введите рост юноши: "))
    heights.append(height)

# проходим по списку ростов и увеличиваем счетчик, если рост меньше 165 см
for height in heights:
    if height < 165:
        count += 1

print("Количество юношей с ростом менее 165 см: ", count)
