naturals = [] # создаем пустой список для натуральных чисел

for i in range(100, 1000): # проходим по числам из интервала [100, 999]
  if i % 10 == 7 and i % 9 == 0: # проверяем, что число оканчивается на 7, кратно 9
    naturals.append(i) # добавляем число в список
    if len(naturals) == 10: # если в списке уже 10 чисел, выходим из цикла
      break

print(naturals) # выводим список на экран
