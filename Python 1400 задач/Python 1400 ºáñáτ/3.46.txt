# Определяем k 
k = 10  # пример

# Определяем условие k-ой цифры 
if k % 3 == 0:
    num = str(100 + k // 3)
    print(num[k % 3 - 1])  # выводим k-ую цифру
elif k % 3 == 1:
    num = str(100 + k // 3 + 1)
    print(num[0])  # выводим k-ую цифру
else:
    num = str(100 + k // 3 + 1)
    print(num[2])  # выводим k-ую цифру
