n = int(input("Введите натуральное число: "))
m = int(input("Введите количество последних цифр для суммирования: "))

if n < 10**(m-1):
    print("Ошибка: введенное число меньше, чем", m, "- значное.")
else:
    sum_digit = 0
    for i in range(m):
        sum_digit += n % 10
        n //= 10
    print("Сумма последних", m, "цифр равна", sum_digit)
