
# создаем массив с данными о количестве осадков
precipitation = [0, 3.2, 0, 0, 0, 1.5, 0, 0, 0, 0, 0, 0]

# инициализируем переменную для хранения количества дней без осадков
no_precipitation_days = 0

# проходим по элементам массива и проверяем, равен ли элемент нулю
for p in precipitation:
    if p == 0:
        no_precipitation_days += 1

# выводим результат на экран
print("Количество дней без осадков:", no_precipitation_days)
