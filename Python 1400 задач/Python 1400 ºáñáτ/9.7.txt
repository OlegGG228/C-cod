# создание матрицы оценок учеников
grades = [
    [4, 5, 3],
    [5, 4, 4],
    [3, 2, 4],
    [4, 3, 5],
    [5, 4, 4],
    [2, 5, 3],
    [3, 4, 5],
    [4, 5, 4],
    [3, 4, 2],
    [5, 5, 5],
    [2, 3, 4],
    [5, 4, 5],
    [4, 5, 3],
    [3, 3, 4],
    [4, 4, 4]
]

# рассчет среднего балла каждого ученика и вывод результата на экран
for i in range(len(grades)):
    avg = sum(grades[i]) / len(grades[i])
    print("Ученик", i+1, "- средний балл:", round(avg, 2))
