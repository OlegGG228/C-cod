# а) сумма кубов всех целых чисел от 25 до 40
sum = 0
for i in range(25, 41):
    sum += i ** 3
print("Сумма кубов всех целых чисел от 25 до 40:", sum)

# б) сумма квадратов всех целых чисел от a до 50
a = int(input("Введите число а: "))
sum = 0
for i in range(a, 51):
    sum += i ** 2
print("Сумма квадратов всех целых чисел от", a, "до 50:", sum)

# в) сумма квадратов всех целых чисел от 1 до п
p = int(input("Введите число п: "))
sum = 0
for i in range(1, p+1):
    sum += i ** 2
print("Сумма квадратов всех целых чисел от 1 до", p, ":", sum)

# г) сумма квадратов всех целых чисел от а до b
a = int(input("Введите число а: "))
b = int(input("Введите число b (b > а): "))
sum = 0
for i in range(a, b+1):
    sum += i ** 2
print("Сумма квадратов всех целых чисел от", a, "до", b, ":", sum)
