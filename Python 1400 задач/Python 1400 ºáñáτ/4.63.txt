a = int(input("Введите внутренний размер конверта по ширине в мм: "))
b = int(input("Введите внутренний размер конверта по высоте в мм: "))
e = int(input("Введите ширину открытки в мм: "))
n = int(input("Введите высоту открытки в мм: "))

if (a-2 >= e) and (b-2 >= n):
    print("Открытка поместится в конверт")
else:
    print("Открытка не поместится в конверт")
