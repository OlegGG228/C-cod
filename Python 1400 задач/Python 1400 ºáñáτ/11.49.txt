rainfall = [4, 2, 5, 0, 3, 1, 6, 3, 2, 1, 5, 2] # пример массива с количеством осадков за каждый день февраля

avg_rainfall = sum(rainfall) / len(rainfall) # среднедневное количество осадков

print(f"Среднедневное количество осадков в феврале: {avg_rainfall}") # вывод среднедневного количества осадков в феврале
