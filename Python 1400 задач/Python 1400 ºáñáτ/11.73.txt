heights = [175, 180, 160, 170, 165, 175, 165, 185, 170, 180, 163, 175, 170, 168, 172, 169, 177, 190, 166, 182, 176, 171]
r = int(input("Введите значение r: "))

print("Количество учеников, рост которых не превышает значения r:", count)
