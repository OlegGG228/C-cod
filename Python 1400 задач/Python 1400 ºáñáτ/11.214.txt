# Пример списка данных о сумме очков футбольных команд
scores = {'Команда А': 18, 'Команда Б': 15, 'Команда В': 12, 'Команда Г': 9, 'Команда Д': 6, 'Команда Е': 3}

# Создание списка мест команд в чемпионате
places = ['1 место: Команда А', '2 место: Команда Б', '3 место: Команда В', '4 место: Команда Г', '5 место: Команда Д', '6 место: Команда Е']

# Проверка соответствия списка мест списку данных о сумме очков
if [key for key in sorted(scores, key=scores.get, reverse=True)] == [place.split(': ')[1] for place in places]:
    print('Список команд в соответствии с занятыми местами в чемпионате')
else:
    print('Список команд не соответствует занятым местам в чемпионате')
