results = [60, 50, 55, 62, 70, 68, 78, 75, 72, 61, 63, 59, 58, 64, 66, 69, 73, 71, 67, 65, 56, 57, 74, 77, 76]

winner_index = 0 # индекс победителя, начинаем с первого спортсмена

for i in range(1, len(results)): # пробегаемся по всем элементам списка результатов
    if results[i] > results[winner_index]: # если очередной результат больше, чем результат победителя
        winner_index = i # обновляем индекс победителя

print(f"Победитель гонки имеет результат {results[winner_index]}")
