word = input("Введите слово: ") # просим пользователя ввести слово
if word[0] == word[-1]: # проверяем, является ли первая буква слова равной последней
    print("Слово начинается и заканчивается на одну и ту же букву")
else:
    print("Слово не начинается и не заканчивается на одну и ту же букву")
