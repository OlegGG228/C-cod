
# Запрос размеров стола и кости домино
a = int(input("Введите размер стола а: "))
b = int(input("Введите размер стола b: "))
c = int(input("Введите размер кости домино c: "))
d = int(input("Введите размер кости домино d: "))
e = int(input("Введите размер кости домино е: "))

# Рассчитываем количество костей домино, которые можно разместить по ширине и длине стола
count_width = (a // c) * (b // d)
count_length = (a // d) * (b // c)

# Если размер е больше размера d, то можно рассчитать количество домино, которые можно разместить "стоя"
if e > d:
    count_height = (a // e) * (b // c)
    # Выводим максимальное количество костей домино, которое можно разместить на столе
    print("Максимальное количество костей домино: ", max(count_width, count_length, count_height))
else:
    # Выводим максимальное количество костей домино, которое можно разместить на столе
    print("Максимальное количество костей домино: ", max(count_width, count_length))
