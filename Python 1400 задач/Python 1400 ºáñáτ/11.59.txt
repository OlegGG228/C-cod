# а) 
```
arr = [1, 9, 15, 21, 3, 7, 20]
sum_a = 0
for i in arr:
    if i <= 20:
        sum_a += i
print("Сумма элементов, значение которых не превышает 20, равна:", sum_a)
```

```
arr = [1, 9, 15, 21, 3, 7, 20]
a = 10
sum_b = 0
for i in arr:
    if i > a:
        sum_b += i
print("Сумма элементов, больших", a, "равна:", sum_b)
