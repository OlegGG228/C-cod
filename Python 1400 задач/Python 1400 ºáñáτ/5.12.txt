p0 = 1.29 # плотность на уровне моря
z = 1.25 * 10 ** (-4) # коэффициент
step = 100 # шаг для высоты

for h in range(0, 1001, step): # перебираем высоту от 0 до 1000 м с шагом step
    p = p0 * (2.71828 ** (-z * h)) # формула рассчета плотности по высоте
    print("Высота: {} м, плотность: {:.2f} кг/м3".format(h, p)) # выводим результат с двумя знаками после запятой
