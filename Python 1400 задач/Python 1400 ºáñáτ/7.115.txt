m = int(input("Введите количество чисел: "))
n = int(input("Введите число, на которое будем проверять кратность: "))
sum_a = 0
count_a = 0

for i in range(m):
    a = int(input())
    if a % n == 0:
        sum_a += a
        count_a += 1

if count_a > 0:
    print("Среднее арифметическое чисел, кратных", n, "равно", sum_a / count_a)
else:
    print("Среднее арифметическое чисел, кратных", n, "равно 0, т.к. таких чисел не найдено.")
