substring = input("Введите подстроку: ")
t = int(input("Введите номер символа, с которого начинается подстрока: "))
n = int(input("Введите номер символа, на котором заканчивается подстрока: "))

if int(substring[t:n+1]) == 666:
    print("Данная подстрока образует число 666")
else:
    print("Данная подстрока не образует число 666")
