#a) для разных решений:

s = int(input("Введите площадь: "))

for a in range(1, s + 1):
    if s % a == 0:
        b = s // a
        print("Прямоугольник со сторонами", a, "и", b)

#b) для совпадающих решений:

s = int(input("Введите площадь: "))

for a in range(1, int(s ** 0.5) + 1):
    if s % a == 0:
        b = s // a
        if a <= b:
            print("Прямоугольник со сторонами", a, "и", b)
