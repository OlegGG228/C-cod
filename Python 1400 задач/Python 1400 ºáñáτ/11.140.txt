arr = [10.7, 9.5, 11.2, 10.8, 12.1, 9.9, 11.1, 11.5, 10.3, 
       12.3, 9.8, 11.7, 10.5, 12.2, 10.9, 11.6, 10.4, 11.3, 
       9.4, 12.4, 10.1, 11.9, 10.6, 12.0, 11.0, 9.7, 11.4, 
       10.2, 11.8, 9.6]

max1 = max2 = float('-inf')
for i in arr:
    if i > max1:
        max2 = max1
        max1 = i
    elif i > max2:
        max2 = i
        
print("Первое и второе места:", max1, max2)
