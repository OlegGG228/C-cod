# Добавляем рост двух новых учеников в упорядоченный массив
arr = [185, 180, 175, 170, 165, 160, 155, 150, 145, 140, 135, 130, 125, 120, 115, 110, 105, 100, 95, 90, 85, 80, 75, 70, 65]
i1 = 6
i2 = 18
new_heights = [166, 173]
arr.insert(i1, new_heights[0])
arr.insert(i2, new_heights[1])
print(arr)  # [185, 180, 175, 170, 165, 160, 166, 155, 150, 145, 140, 135, 130, 125, 120, 115, 110, 105, 100, 95, 90, 85, 80, 75, 70, 65, 173]
