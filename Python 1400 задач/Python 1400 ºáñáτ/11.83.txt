# Создаем массив
arr = [4, 1, 6, 8, 2, 5, 3, 9]

# Создаем переменную для хранения количества элементов, удовлетворяющих условие
count = 0

# Обходим все элементы массива, начиная с второго и заканчивая предпоследним
for i in range(1, len(arr) - 1):
    # Если текущий элемент больше предыдущего и следующего, увеличиваем счетчик на 1
    if arr[i] > arr[i-1] and arr[i] > arr[i+1]:
        count += 1

# Выводим количество элементов, удовлетворяющих условие
print("Количество элементов массива, которые больше своих 'соседей':", count)
