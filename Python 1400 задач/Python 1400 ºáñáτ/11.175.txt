# Вставляем число п между всеми соседними элементами, имеющими одинаковый знак
arr = [-5, 2, 3, -3, -3, 0, 1, -1, -9, 6, -6, 1]
p = 7
i = 0
while i < len(arr)-1:
    if (arr[i] > 0 and arr[i+1] > 0) or (arr[i] < 0 and arr[i+1] < 0):
        arr.insert(i+1, p)
        i += 2
    else:
        i += 1
print(arr)  # [-5, 2, 3, -3, 7, -3, 0, 1, -1, -9, 6, 7, -6, 1]
