precipitation = [25, 10, 15, 20, 30, 5, 10, 40, 35, 20, 15, 5] # список количества осадков
max_precipitation = max(precipitation) # находим максимальное количество осадков

for i in range(len(precipitation)-1, -1, -1): # проходим по списку в обратном порядке
    if precipitation[i] == max_precipitation: # находим последний день с максимальным количеством осадков
        print("День месяца с наибольшим количеством осадков:", i+1) # выводим число дня месяца
        break # останавливаем цикл после нахождения последнего дня с максимальным количеством осадков
