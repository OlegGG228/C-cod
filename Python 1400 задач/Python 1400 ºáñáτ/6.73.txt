g = int(input("Введите первый член геометрической прогрессии: "))
z = int(input("Введите знаменатель геометрической прогрессии: "))
t = int(input("Введите число для проверки: "))

if t == g:
    print(t, "является первым членом геометрической прогрессии")
elif (t / g) % z == 0:
    print(t, "является членом геометрической прогрессии")
else:
    print(t, "не является членом геометрической прогрессии")
