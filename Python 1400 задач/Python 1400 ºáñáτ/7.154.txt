sequence = input("Введите последовательность чисел через пробел: ").split() # ввод последовательности

count = 1 # счетчик подряд идущих одинаковых элементов
max_count = 1 # счетчик наибольшего количества подряд идущих одинаковых элементов

# проверяем каждый элемент соседний с ним
for i in range(1, len(sequence)):
    if sequence[i] == sequence[i-1]: # если элементы равны
        count += 1 # увеличиваем счетчик
        if count > max_count: # если текущее количество подряд идущих одинаковых элементов больше максимального
            max_count = count # меняем значение максимального счетчика
    else: # если элементы не равны
        count = 1 # обнуляем счетчик

print("Наибольшее количество подряд идущих одинаковых элементов в последовательности:", max_count)
