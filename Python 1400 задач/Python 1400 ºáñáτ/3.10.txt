k = int(input("Введите номер дня в году (1 < k < 365): "))
d = int(input("Введите номер дня недели, на которую приходится 1 января (1 - понедельник, 2 - вторник, ..., 7 - воскресенье): "))

weekday = (d + k - 2) % 7 + 1

if weekday in (1, 2, 3, 4, 5, 6):
    p = weekday
else:
    p = 0

print("Для дня", k, "в году с днём недели", d, "приходится день недели", p)
