m = int(input("Введите число m: ")) # вводим число m
arr = [5, 2, 7, 4, 8, 3, 9, 1, 6] # исходный массив
sum = 0 # переменная для хранения суммы элементов меньше m
count = 0 # переменная для хранения количества элементов меньше m
for i in arr:
    if i < m: # если элемент меньше m
        sum += i # добавляем его к сумме
        count += 1 # увеличиваем счетчик
if count > 0: # если такие элементы есть
    average = sum / count # вычисляем среднее арифметическое
    print("Среднее арифметическое элементов меньше", m, "равно", average)
else:
    print("Нет элементов меньше", m) # если таких элементов нет, выводим сообщение об этом
