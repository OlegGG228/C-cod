
a = int(input("Введите размер стола a: "))
b = int(input("Введите размер стола b: "))
c = int(input("Введите размер прямоугольника c: "))
d = int(input("Введите размер прямоугольника d: "))

num_x = a // c
num_y = b // d

if num_x * (b // d) > (a // c) * num_y:
    print("Установка прямоугольников длинной стороной параллельно длинной стороне стола позволяет разместить большее число прямоугольников.")
else:
    print("Установка прямоугольников длинной стороной параллельно короткой стороне стола позволяет разместить большее число прямоугольников.")
