a2 = int(input("Введите первую цифру десятков первого числа: "))
a1 = int(input("Введите первую цифру единиц первого числа: "))
b2 = int(input("Введите первую цифру десятков второго числа: "))
b1 = int(input("Введите первую цифру единиц второго числа: "))

diff = (a2 * 10 + a1) - (b2 * 10 + b1)
diff_a2 = diff // 10  # Получаем первую цифру десятков из разности
diff_a1 = diff % 10  # Получаем первую цифру единиц из разности

print("Цифры числа-разности: ", diff_a2, diff_a1)
