a = [1, 4, 3, 8, 5, 8, 6, 8, 7] # пример исходной последовательности
A = 0 # значение A

print("Исходная последовательность:", a)

max_elements = a.count(8) # количество максимальных элементов в исходной последовательности
if max_elements != 4:
    print("Ошибка: количество максимальных элементов должно быть равно 4")
else:
    a.append(A) # добавляем новый элемент в конец последовательности

    max_elements = a.count(8) # количество максимальных элементов в новой последовательности при A = 0
    print("Количество максимальных элементов в новой последовательности при A = 0:", max_elements)

    A = 8 # меняем значение A на 8

    a[-1] = A # заменяем последний элемент на A

    max_elements = a.count(8) # количество максимальных элементов в новой последовательности при A = 8
    print("Количество максимальных элементов в новой последовательности при A = 8:", max_elements)
