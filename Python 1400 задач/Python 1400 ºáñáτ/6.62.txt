num = int(input("Введите натуральное число: "))

# а) получаемое при прочтении его цифр справа налево
reverse_num = int(str(num)[::-1])
print("Число, полученное при прочтении его цифр справа налево:", reverse_num)

# б) получаемое в результате приписывания по двойке в начало и конец записи исходного числа
new_num = int("2" + str(num) + "2")
print("Число, полученное в результате приписывания по двойке в начало и конец записи исходного числа:", new_num)

# в) получаемое в результате удаления из него всех цифр а
a = input("Введите цифру, которую нужно удалить: ")
new_num = int(str(num).replace(a, ""))
print("Число, полученное в результате удаления из него всех цифр", a, ":", new_num)

# г) получаемое из исходного перестановкой его первой и последней цифр
first_digit = str(num)[0]
last_digit = str(num)[-1]
new_num = int(last_digit + str(num)[1:-1] + first_digit)
print("Число, полученное из исходного перестановкой его первой и последней цифр:", new_num)

# д) образованное из исходного приписыванием к нему такого же числа
new_num = int(str(num) + str(num))
print("Число, образованное из исходного приписыванием к нему такого же числа:", new_num)
