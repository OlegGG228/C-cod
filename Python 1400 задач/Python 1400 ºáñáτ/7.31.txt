sum_ = 0  # переменная для хранения суммы чисел
count = 0  # переменная для хранения количества чисел
while True:  # бесконечный цикл
    num = int(input())  # считываем число
    if num < 0:  # если число отрицательное, прерываем цикл
        break
    sum_ += num  # добавляем число к сумме
    count += 1  # увеличиваем счетчик чисел
if count > 0:  # если было введено хотя бы одно положительное число
    avg = sum_ / count  # вычисляем среднее арифметическое
    print(avg)  # выводим результат
else:  # если не было введено ни одного положительного числа
    print("Среднее арифметическое не определено")
