weight_list = [150, 200, 100, 50] # список масс грузов
max_weight = 500 # грузоподъемность автомобиля
total_weight = sum(weight_list) # общая масса всех грузов

if total_weight <= max_weight:
   print("Общая масса грузов не превышает грузоподъемность автомобиля")
else:
   print("Общая масса грузов превышает грузоподъемность автомобиля")
