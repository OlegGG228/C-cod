a = float(input("Введите число a: "))
b = float(input("Введите число b: "))
c = float(input("Введите число c: "))

if a == b or a == c or b == c:
    print("Среди введенных чисел есть хотя бы одна пара равных между собой чисел")
else:
    print("Среди введенных чисел нет пар равных между собой чисел")
