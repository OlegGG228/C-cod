numbers = [2.5, -3.7, 1.2, 0, 4.8, -6.1, 3.9, -2.2, 5.5, 7.6, -8.3, 9.1, 0.2, -1.6, 3.4]

negative_index = -1 # порядковый номер первого отрицательного числа

for i in range(len(numbers)):
    if numbers[i] < 0:
        negative_index = i + 1 # увеличиваем на 1, так как порядковый номер начинается с 1
        break
    
if negative_index > -1:
    print("Первое отрицательное число находится на позиции", negative_index)
else:
    print("В последовательности нет отрицательных чисел.")
