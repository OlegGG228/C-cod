arr = [120, 110, 130, 140, 115, 125, 135, 150, 122, 128, 
       142, 153, 160, 118, 132, 124, 146, 152, 137, 129, 
       119, 148, 121, 127, 133, 143, 136, 131, 139, 145, 
       141, 147, 149, 126, 144, 123, 138, 116, 151, 134]
