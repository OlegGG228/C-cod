# создаем функцию для нахождения всех делителей числа
def divisors(n):
    divs = []
    for i in range(1, n+1):
        if n % i == 0:
            divs.append(i)
    return divs

# инициализируем переменную для подсчета общей суммы всех делителей
total_sum = 0

# итерируемся по всем числам от 50 до 70
for i in range(50, 71):
    divs = divisors(i) # находим все делители данного числа
    div_sum = sum(divs) # считаем сумму делителей
    print("Число {} имеет делители: {}, их сумма равна {}".format(i, divs, div_sum))
    total_sum += div_sum # добавляем текущую сумму делителей к общей

print("Сумма всех делителей чисел от 50 до 70 равна", total_sum)
