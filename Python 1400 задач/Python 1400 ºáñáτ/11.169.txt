# Добавляем рост нового ученика в упорядоченный массив
arr = [180, 175, 170, 165, 160, 155, 150, 145, 140, 135, 130, 125, 120, 115, 110]
i = 4  # новый ученик на 5-ой место
new_height = 170
arr.insert(i, new_height)
print(arr)  # [180, 175, 170, 165, 170, 160, 155, 150, 145, 140, 135, 130, 125, 120, 115, 110]
