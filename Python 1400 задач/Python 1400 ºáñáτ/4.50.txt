num = int(input("Введите двузначное число: "))
if "4" in str(num) or "7" in str(num):
    print("В число входит цифра 4 или 7")
else:
    print("В число не входит цифра 4 или 7")
if "3" in str(num) or "6" in str(num) or "9" in str(num):
    print("В число входит цифра 3, 6 или 9")
else:
    print("В число не входит цифра 3, 6 или 9")
