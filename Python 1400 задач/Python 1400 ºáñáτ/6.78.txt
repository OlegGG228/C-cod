number = input("Введите число: ")
flag = True # флаг для проверки упорядоченности
for i in range(len(number)-1):
  if int(number[i]) < int(number[i+1]): # если цифры не упорядочены по убыванию, то число не подходит
    flag = False
    break

if flag:
  print("Последовательность цифр при просмотре справа налево упорядочена по возрастанию")
else:
  print("Последовательность цифр при просмотре справа налево не упорядочена по возрастанию")
