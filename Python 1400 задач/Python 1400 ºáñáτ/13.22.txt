schools = [{"Количество учащихся": 100, "Тип": "Школа"}, {"Количество учащихся": 80, "Тип": "Техникум"},
           {"Количество учащихся": 120, "Тип": "Школа"}, {"Количество учащихся": 90, "Тип": "Училище"}] # и т.д.

total_students = 0

for school in schools:
  if school["Тип"] == "Школа":
    total_students += school["Количество учащихся"]

print("Общее число учащихся в школах:", total_students)
