p = int(input("Введите натуральное число p: "))
sequence = list(map(int, input("Введите последовательность чисел через пробел: ").split()))

# Используем переменную max_len для хранения наибольшей длины отрезка четных чисел
max_len = 0 
# Используем переменную current_len для хранения текущей длины отрезка четных чисел
current_len = 0 

for num in sequence:
    if num % 2 == 0: # Если число четное, увеличиваем длину текущего отрезка
        current_len += 1
    else: # Если число нечетное, проверяем, является ли текущий отрезок наибольшим и обнуляем текущую длину
        if current_len > max_len:
            max_len = current_len
        current_len = 0 # Обнуляем текущую длину

# Проверяем, является ли текущий отрезок наибольшим после прохода по всей последовательности
if current_len > max_len:
    max_len = current_len
    
print("Наибольшая длина отрезка четных чисел:", max_len)
