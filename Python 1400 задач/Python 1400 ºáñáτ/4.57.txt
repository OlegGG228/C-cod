a = int(input("Введите целое неотрицательное число a: "))
b = int(input("Введите целое положительное число b: "))
c = int(input("Введите число c: "))
d = int(input("Введите число d: "))

if a < 0:
    print("Ошибка: число 'a' должно быть неотрицательным")
elif b <= 0:
    print("Ошибка: число 'b' должно быть положительным")
else:
    remainder = a % b
    if remainder == c or remainder == d:
        print("Равенство выполняется")
    else:
        print("Равенство не выполняется")
