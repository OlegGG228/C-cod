grades = [4, 3, 5, 2, 4, 3, 4, 5, 2, 3, 2, 3, 4, 5, 2, 4, 3, 4, 5, 2, 3, 4, 5, 2, 3] # оценки учеников
failing_students = 0 # переменная для подсчета числа неуспевающих студентов

for grade in grades: # для каждой оценки
    if grade < 3: # если оценка ниже 3
        failing_students += 1 # увеличиваем счетчик неуспевающих студентов

print("Количество неуспевающих по химии учеников: ", failing_students) # выводим результат
