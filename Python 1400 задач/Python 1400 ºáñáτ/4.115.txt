nums = [5, -2, 3, -7, 10, 1]    # пример списка чисел
positive_sum = 0                # начинаем сумму нулем

for num in nums:
    if num > 0:                # если число положительное
        positive_sum += num    # добавляем его к сумме

print("Сумма положительных чисел:", positive_sum)  # выводим результат
