# а) Копирование в том же порядке расположения

# исходный массив
arr1 = [1, 2, 3, 4, 5]

# создаем новый массив такой же длины, как исходный
arr2 = [0] * len(arr1)

# копируем элементы в новый массив
for i in range(len(arr1)):
    arr2[i] = arr1[i]

# выводим оба массива на экран
print("Исходный массив:", arr1)
print("Новый массив в том же порядке:", arr2)


# б) Копирование в обратном порядке расположения

# исходный массив
arr3 = [1, 2, 3, 4, 5]

# создаем новый массив такой же длины, как исходный
arr4 = [0] * len(arr3)

# копируем элементы в новый массив в обратном порядке
for i in range(len(arr3)):
    arr4[i] = arr3[-i-1]

# выводим оба массива на экран
print("Исходный массив:", arr3)
print("Новый массив в обратном порядке:", arr4)
