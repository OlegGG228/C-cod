# определяем функцию, которая возвращает количество клеток через n часов
def count_cells(n):
    cells = 1 # начальное количество клеток
    for i in range(n//3): # каждые 3 часа клетка делится на 2, поэтому бегаем по количеству 3-х часовых промежутков
        cells *= 2 # умножаем количество клеток на 2, т.к. каждая клетка делится на 2
    return cells
