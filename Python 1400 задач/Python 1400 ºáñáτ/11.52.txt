import math

array = [1, 2, 3, 4, 5]  # заданный массив
sum_array = sum(array)  # сумма элементов массива
sum_squares_array = sum([i**2 for i in array])  # сумма квадратов элементов массива

if sum_array % 2 == 0:
    print("Сумма элементов массива является четным числом")
else:
    print("Сумма элементов массива является нечетным числом")

if math.floor(math.log10(sum_squares_array)) + 1 == 5:
    print("Сумма квадратов элементов массива является пятизначным числом")
else:
    print("Сумма квадратов элементов массива не является пятизначным числом")
