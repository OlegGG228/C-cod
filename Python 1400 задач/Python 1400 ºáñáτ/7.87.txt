d_list = input("Введите целые числа через пробел: ").split()

count = 0
for d in d_list:
    if int(d) > 0 and int(d) % 3 == 0:
        count += 1
        
if count == 0:
    print("В заданной последовательности нет положительных чисел, кратных трем")
elif count == len(d_list):
    print("Все положительные числа в заданной последовательности кратны трем")
else:
    print("Не все положительные числа в заданной последовательности кратны трем")
