grades = [7, 9, 8, 10, 9, 9, 8, 6, 7, 8, 9, 8, 7, 9, 10, 9, 8, 9]

# Вычисляем среднюю оценку по обязательной программе
comp_avg = sum(grades[:6]) / 6

# Вычисляем среднюю оценку по произвольной программе
free_avg = sum(grades[6:]) / 12

# Сравниваем средние оценки и выводим результат
if comp_avg > free_avg:
    print("Спортсмен показал лучший результат по обязательной программе")
else:
    print("Спортсмен показал лучший результат по произвольной программе")
