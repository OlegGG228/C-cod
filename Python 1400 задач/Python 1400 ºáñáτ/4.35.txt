number = input('Введите четырёхзначное число: ')
# проверяем, является ли введенная строка четырехзначным числом
if not number.isdigit() or len(number) != 4:
    print('Некорректный ввод')
else:
    # преобразуем строку в список цифр
    digits = [int(d) for d in number]
    # проверяем условия
    if digits[0] + digits[1] == digits[2] + digits[3]:
        print('Сумма двух первых цифр равна сумме двух последних')
    else:
        print('Сумма двух первых цифр не равна сумме двух последних')
        
    if sum(digits) % 3 == 0:
        print('Сумма цифр кратна трём')
    else:
        print('Сумма цифр не кратна трём')
        
    if (digits[0] * digits[1] * digits[2] * digits[3]) % 4 == 0:
        print('Произведение цифр кратно четырем')
    else:
        print('Произведение цифр не кратно четырем')
        
    a = 7 # произвольное число для проверки четвертого варианта
    if (digits[0] * digits[1] * digits[2] * digits[3]) % a == 0:
        print(f'Произведение цифр кратно числу {a}')
    else:
        print(f'Произведение цифр не кратно числу {a}')
