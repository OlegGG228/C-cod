number = int(input("Введите трехзначное число: "))
a = number // 100 # находим первую цифру
b = (number // 10) % 10 # находим вторую цифру
c = number % 10 # находим третью цифру
new_number = b * 100 + a * 10 + c # меняем местами первую и вторую цифры
print("Число после перестановки цифр:", new_number)
