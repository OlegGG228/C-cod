h = int(input("Введите часы (от 1 до 12): "))
m = int(input("Введите минуты (от 0 до 59): "))

# Расчет угла между стрелками в градусах
angle = abs(0.5 * ((60 * h) - (11 * m)))
if angle > 180:
    angle = 360 - angle

# Расчет времени до совпадения
if angle == 0:
    time_to_match = 720 - (h * 60 + m)
    print("До совпадения стрелок осталось", time_to_match, "минут")

# Расчет времени до перпендикулярности
elif angle == 90:
    time_to_perpendicular = 360 - (h * 60 + m)
    print("До перпендикулярности стрелок осталось", time_to_perpendicular, "минут")

else:
    print("Стрелки не совпадут и не будут перпендикулярны")
