students = [] # создаем пустой список для хранения количества студентов по курсам и группам

# вводим информацию по каждой группе
for i in range(5): # цикл по курсам от 1 до 5
    course = [] # создаем пустой список для текущего курса
    print("Введите количество студентов для курса", i+1) # выводим запрос на ввод количества студентов для текущего курса
    for j in range(6): # цикл по группам от 1 до 6
        group = int(input("Группа "+str(j+1)+": ")) # получаем от пользователя количество студентов для текущей группы
        course.append(group) # добавляем число студентов для текущей группы в список для текущего курса
    students.append(course) # добавляем список для текущего курса в общий список

# а) находим курс с наименьшим числом студентов
min_course = 0 # номер курса с наименьшим числом студентов
min_students = sum(students[0]) # количество студентов на первом курсе
for i in range(1, 5): # цикл по курсам от 2 до 5
    students_count = sum(students[i]) # количество студентов на текущем курсе
    if students_count < min_students: # если на текущем курсе студентов меньше, чем на предыдущих
        min_course = i # запоминаем номер курса с наименьшим числом студентов
        min_students = students_count # запоминаем количество студентов на этом курсе

print("Меньше всего студентов на", min_course+1, "курсе")

# б) находим самую малочисленную группу по всем курсам
min_group = [0, 0] # номер курса и номер группы самого малочисленной группы
min_students = students[0][0] # количество студентов в первой группе первого курса
for i in range(5): # цикл по курсам от 1 до 5
    for j in range(6): # цикл по группам от 1 до 6
        students_count = students[i][j] # количество студентов в текущей группе текущего курса
        if students_count < min_students: # если в текущей группе студентов меньше, чем в предыдущих
            min_group = [i+1, j+1] # запоминаем номер курса и номер группы самой малочисленной группы
            min_students = students_count # запоминаем количество студентов в этой группе

print("Самая малочисленная группа – группа", min_group[1], "на", min_group[0], "курсе")

# в) находим самую малочисленную группу для каждого курса
for i in range(5): # цикл по курсам от 1 до 5
    min_group = 0 # номер самой малочисленной группы на текущем курсе
    min_students = students[i][0] # количество студентов в первой группе текущего курса
    for j in range(1, 6): # цикл по группам от 2 до 6
        students_count = students[i][j] # количество студентов в текущей группе текущего курса
        if students_count < min_students: # если в текущей группе студентов меньше, чем в предыдущих
            min_group = j+1 # запоминаем номер самой малочисленной группы на текущем курсе
            min_students = students_count # запоминаем количество студентов в этой группе
    print("Самая малочисленная группа на", i+1, "курсе – группа", min_group)
