# создаем словарь элементов
elements = {
    "медь": "Cu",
    "кислород": "O",
    "железо": "Fe",
    "уран": "U",
    "серебро": "Ag",
    "свинец": "Pb",
    "кобальт": "Co",
    "силикон": "Si",
    "серебро": "Ag"
}

# запрашиваем обозначение элемента
symbol = input("Введите обозначение элемента: ")

# проверяем, есть ли заданное обозначение элемента в словаре
found = False
for name, sym in elements.items():
    if sym == symbol:
        found = True
        break

if found:
    print("Обозначение элемента найдено")
else:
    print("Обозначение элемента не найдено")
