# Пример решения задачи:

heights = [175, 162, 180, 155, 170] # список ростов

max_height = max(heights) # находим максимальный рост
min_height = min(heights) # находим минимальный рост

difference = max_height - min_height # вычисляем разницу

print("Разница между ростом самого высокого и самого низкого человека:", difference) # выводим результат
