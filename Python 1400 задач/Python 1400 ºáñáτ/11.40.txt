# задаем исходный массив
array = [5, 9, 16, 25, 36]

# a) квадратный корень из любого элемента массива
index = int(input("Введите индекс элемента массива, из которого нужно извлечь корень: "))
if index >= len(array):
    print("Индекс вне диапазона массива")
else:
    print("Квадратный корень из элемента", array[index], "равен", round(array[index]**0.5, 2))

# б) среднее арифметическое двух любых элементов массива
index1 = int(input("Введите индекс первого элемента массива: "))
index2 = int(input("Введите индекс второго элемента массива: "))
if index1 >= len(array) or index2 >= len(array):
    print("Индекс вне диапазона массива")
else:
    average = (array[index1] + array[index2]) / 2
    print("Среднее арифметическое элементов", array[index1], "и", array[index2], "равно", round(average, 2))
