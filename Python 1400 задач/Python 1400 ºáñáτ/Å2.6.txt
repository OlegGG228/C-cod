# Решение задачи а)

a = 3.4     # a присваиваем значение 3.4
b = -5.1    # b присваиваем значение -5.1
b = a       # b присваиваем значение переменной a, то есть 3.4
a = b       # a присваиваем значение переменной b, то есть 3.4

print("a =", a) # выводим значение переменной a
print("b =", b) # выводим значение переменной b
