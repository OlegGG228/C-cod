cars = [{"Стоймость": 100000, "Возраст": 5}, {"Стоймость": 150000, "Возраст": 2},
        {"Стоймость": 130000, "Возраст": 7}, {"Стоймость": 140000, "Возраст": 10}] # и т.д.

total_cost = 0
count = 0

for car in cars:
  if car["Возраст"] > 6:
    total_cost += car["Стоймость"]
    count += 1

if count > 0:
  average_cost = total_cost/count
  print("Средняя стоимость автомобилей, возраст которых превышает 6 лет:", average_cost)
else:
  print("Автомобилей, возраст которых превышает 6 лет, нет")
