import math

d = 10 # внутренний диаметр внутреннего шара в см
t = 0.5 # толщина стенки шара (5 мм)
n = 12 # количество шаров

r = (d/2) # радиус внутреннего шара в см
V_total = ((4/3) * math.pi * (r + n*t)**3) - ((4/3) * math.pi * r**3) # общий объем в см^3
V_liter = V_total / 1000 # приводим к литрам

print(V_liter) # выводим суммарный объем в литрах
