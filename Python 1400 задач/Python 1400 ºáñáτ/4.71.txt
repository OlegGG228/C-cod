import math

# константы:
g = 9.8 # ускорение свободного падения

# входные данные:
a_deg = float(input("Угол вылета снаряда (градусы): "))
v0 = float(input("Начальная скорость снаряда (м/c): "))
R = float(input("Расстояние до цели вдоль земной поверхности (м): "))
P = float(input("Высота цели над стволом орудия (м): "))

# перевод градусов в радианы:
a_rad = a_deg*math.pi/180

# время полета снаряда до падения на землю:
t = 2*v0*math.sin(a_rad)/g

# координаты цели в момент времени t:
x = v0*t*math.cos(a_rad)
y = v0*t*math.sin(a_rad) - 0.5*g*t**2 + P

# проверка попадания в цель:
if abs(x - R) <= 0.1 and y >= 0:
    print("Снаряд поразит цель!")
else:
    print("Снаряд не достигнет цель.")
