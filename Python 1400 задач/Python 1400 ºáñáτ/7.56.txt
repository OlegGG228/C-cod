p = int(input("Введите натуральное число p: "))
c = [] # список для хранения целых чисел c2, ..., cp

for i in range(2, p+1):
    c_i = int(input(f"Введите целое число c{i}: "))
    c.append(c_i)

t = int(input("Введите целое число t: "))
r = int(input("Введите целое число r: "))

s = 0 # переменная для хранения суммы чисел с, которые не превышают t

for c_i in c:
    if c_i <= t:
        s += c_i

if s % r == 0:
    print(f"Сумма чисел, не превышающих t, кратна числу r: {s}")
else:
    print("Сумма чисел, не превышающих t, не кратна числу r")
