array = [3, -5, 6, -2, 8, 0, -1, 4, -9, 7]

positive = 0 # переменная для хранения количества положительных элементов 
negative = 0 # переменная для хранения количества отрицательных элементов 

for i in array: # проходим по всем элементам массива 
    if i > 0: # если элемент положительный, увеличиваем счетчик положительных элементов 
        positive += 1 
    elif i < 0: # если элемент отрицательный, увеличиваем счетчик отрицательных элементов 
        negative += 1 

print(f'Количество положительных элементов в массиве: {positive}') 
print(f'Количество отрицательных элементов в массиве: {negative}')
