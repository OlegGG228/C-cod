a = int(input("Введите числитель дроби: "))
b = int(input("Введите знаменатель дроби: "))

# Находим наибольший общий делитель числителя и знаменателя
def gcd(x, y):
    while y != 0:
        x, y = y, x % y
    return x

# Сокращаем дробь
d = gcd(a, b)
p = a // d
q = b // d

print("Сокращенная дробь: {}/{}".format(p, q))
