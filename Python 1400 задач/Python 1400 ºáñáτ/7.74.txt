n = int(input("Введите количество троек чисел: "))
count = 0 # счетчик количества троек, которые могут быть использованы
for i in range(n):
    a, b, c = map(int, input("Введите тройку чисел a, b, c через пробел: ").split())
    if a + b > c: # условие существования треугольника
        count += 1
print(f"Количество троек, которые могут быть использованы для построения треугольника: {count}")
