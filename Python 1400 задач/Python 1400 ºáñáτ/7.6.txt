
#Создание пустого списка для хранения сопротивлений элементов цепи
resistances = []

#Запрос количества элементов цепи
n = int(input("Введите количество элементов в цепи: "))

#Проход цикла от 1 до n и запрос сопротивления каждого элемента цепи
for i in range(1, n+1):
  resistance = float(input("Введите сопротивление " + str(i) + "-го элемента цепи: "))
  #Добавление сопротивления в список
  resistances.append(resistance)

#Вычисление общего сопротивления цепи с использованием функции sum()
total_resistance = sum(resistances)

#Вывод результата
print("Общее сопротивление цепи равно:", total_resistance)
