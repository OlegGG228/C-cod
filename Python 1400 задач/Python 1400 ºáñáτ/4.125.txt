weight = float(input("Введите вес боксера в кг: "))
if weight <= 60:
    print("Боксер относится к легкому весу")
elif weight <=64:
    print("Боксер относится к первому полусреднему весу")
elif weight <= 69:
    print("Боксер относится к полусреднему весу")
else:
    print("Боксер не может быть отнесен ни к одной из категорий")
