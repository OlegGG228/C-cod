# Исходные данные
a = 425
b = 131

# Счетчик количества квадратов
count = 0

# Пока можно отрезать квадраты со стороной b
while a >= b:
    # Вычисляем количество квадратов, которые можно отрезать
    n = a // b
    # Увеличиваем счетчик на количество отрезанных квадратов
    count += n
    # Вычисляем новые размеры прямоугольника
    a, b = b, a - b * n

# Выводим результаты
print("Исходный прямоугольник был разрезан на", count, "квадратов со стороной", b)
