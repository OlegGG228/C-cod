# вводим данные о количестве осадков
precipitation = [5, 2, 3, 7, 1, 2, 4, 8, 7, 6, 3, 2]

# вычисляем сумму осадков по четным и нечетным числам
even_sum = sum([p for i, p in enumerate(precipitation) if i % 2 == 0])
odd_sum = sum([p for i, p in enumerate(precipitation) if i % 2 != 0])

# проверяем, выпало ли больше осадков по четным числам, чем по нечетным
if even_sum > odd_sum:
    print("Да, по четным числам выпало больше осадков, чем по нечетным")
else:
    print("Нет, по четным числам не выпало больше осадков, чем по нечетным")
