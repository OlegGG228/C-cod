squares = [50, 25, 100, 36, 64] # список площадей квадратов

max_area = max(squares) # находим максимальную площадь квадрата в списке
side = round(max_area**0.5, 2) # находим длину стороны квадрата и округляем до двух знаков после запятой
diag = round(side * (2**0.5), 2) # находим длину диагонали квадрата и округляем до двух знаков после запятой

print("Длина диагонали самого большого квадрата равна:", diag) # выводим результат
