#a) Решение:

distance = 10  # начальный пробег
day = 1  # начальный день

while distance <= 20:  # пока не пробежит больше 20 км
    day += 1  # увеличиваем день
    distance *= 1.1  # увеличиваем пробег на 10 %

print(f"Лыжник пробежит более 20 км на {day}-й день")

#б) Решение:

distance = 10  # начальный пробег
day = 1  # начальный день
total_distance = 10  # начальный суммарный пробег

while total_distance <= 100:  # пока не пробежит больше 100 км
    day += 1  # увеличиваем день
    distance *= 1.1  # увеличиваем пробег на 10 %
    total_distance += distance  # увеличиваем суммарный пробег

print(f"Суммарный пробег превысит 100 км на {day}-й день")
