a = float(input("Введите длину ребра куба: ")) # ввод длины ребра
volume = a ** 3  # вычисление объема куба
area = 6 * a ** 2 # вычисление площади боковой поверхности
print("Объем куба: ", round(volume, 2)) # вывод результата с округлением до 2 знаков после запятой
print("Площадь боковой поверхности: ", round(area, 2)) # вывод результата с округлением до 2 знаков после запятой
