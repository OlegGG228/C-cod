a = int(input("Введите первое двузначное число в формате а2а1: ")) # вводим первое двузначное число
b = int(input("Введите второе двузначное число в формате b2b1: ")) # вводим второе двузначное число
c = a + b # вычисляем сумму a и b
с1 = с % 10 # выделяем цифру единиц суммы
с2 = с // 10 # выделяем цифру десятков суммы
a1 = a % 10 # выделяем цифру единиц числа а
a2 = a // 10 # выделяем цифру десятков числа а
b1 = b % 10 # выделяем цифру единиц числа b
b2 = b // 10 # выделяем цифру десятков числа b
d1 = (a1 + b1 + с1) % 10 # вычисляем цифру единиц числа-результата
d2 = a2 + b2 + с2 + (a1 + b1 + с1) // 10 # вычисляем цифру десятков числа-результата
x = d2 * 10 + d1 # вычисляем число-результат
print("Цифры числа-слагаемого: а1 =", a1, "а2 =", a2, "b1 =", b1, "b2 =", b2)
print("Цифры числа-результата: d1 =", d1, "d2 =", d2)
print("Число-результат:", x)
