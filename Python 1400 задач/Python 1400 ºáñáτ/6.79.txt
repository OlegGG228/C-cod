num = input("Введите натуральное число: ")
ordered = True # флаг упорядоченности

for i in range(1, len(num)):
    if int(num[i]) > int(num[i-1]): # если текущая цифра больше предыдущей
        ordered = False # последовательность неупорядочена
        break # выходим из цикла, дальше проверять не нужно
        
if ordered:
    print("Последовательность цифр упорядочена по неубыванию")
else:
    print("Последовательность цифр НЕ упорядочена по неубыванию")
