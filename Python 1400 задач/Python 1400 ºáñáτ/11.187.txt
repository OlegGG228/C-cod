# а) вставка числа 100 в начало списка
lst = [1, 2, 3, 4, 5]
lst = [100] + lst
print(lst)

# б) вставка заданного числа x в начало списка
x = 99
lst = [1, 2, 3, 4, 5]
lst = [x] + lst
print(lst)

# в) вставка заданного числа x перед элементом с индексом i
x = 77
i = 2
lst = [1, 2, 3, 4, 5]
lst = lst[:i] + [x] + lst[i:]
print(lst)
