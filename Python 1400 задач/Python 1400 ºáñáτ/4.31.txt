num = int(input("Введите трехзначное число: "))
a = num // 100 # первая цифра
b = (num // 10) % 10 # вторая цифра
c = num % 10 # третья цифра

if a > c:
    print("Первая цифра", a, "больше последней", c)
elif c > a:
    print("Последняя цифра", c, "больше первой", a)
else:
    print("Первая и последняя цифры равны")

if a > b:
    print("Первая цифра", a, "больше второй", b)
elif b > a:
    print("Вторая цифра", b, "больше первой", a)
else:
    print("Первая и вторая цифры равны")

if b > c:
    print("Вторая цифра", b, "больше последней", c)
elif c > b:
    print("Последняя цифра", c, "больше второй", b)
else:
    print("Вторая и последняя цифры равны")
