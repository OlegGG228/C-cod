capacity = [12, 5, 8, 20, 10, 15, 25, 30, 40, 50, 60, 70, 80, 90, 100, 110, 120, 130, 140, 150, 160, 170] # вместимость дисков
price = [100, 200, 150, 300, 250, 350, 400, 450, 500, 550, 600, 700, 800, 900, 1000, 1100, 1200, 1300, 1400, 1500, 1600, 1700] # стоимость дисков
s = int(input("Введите цену для поиска: "))
