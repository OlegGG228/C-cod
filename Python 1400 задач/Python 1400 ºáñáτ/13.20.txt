cars = [{"Мощность": 80, "Стоимость": 100000}, {"Мощность": 120, "Стоимость": 150000},
        {"Мощность": 90, "Стоимость": 130000}, {"Мощность": 110, "Стоимость": 140000}] # и т.д.

total_cost = 0

for car in cars:
  if car["Мощность"] > 100:
    total_cost += car["Стоимость"]

print("Общая стоимость автомобилей, у которых мощность двигателя превышает 100 л.с.:", total_cost)
