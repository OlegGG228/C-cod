# Добавляем число а после всех элементов, в которых есть цифра 5
arr = [1, 52, 345, 78, 524, 19, 305, 65, 752, 5, 67, 25, 435, 94]
a = 10
for i in range(len(arr)):
    if "5" in str(arr[i]):
        arr.insert(i+1, a)
print(arr)  # [1, 52, 10, 345, 78, 524, 19, 305, 10, 65, 752, 5, 10, 67, 25, 435, 94]
