n = int(input("Введите сумму: "))  # вводим сумму
ways = 0  # обнуляем счетчик способов
for i in range(n//10 + 1):  # перебор количества 10-рублевых купюр
    for j in range((n - i*10)//5 + 1):  # перебор количества 5-рублевых монет
        for k in range((n - i*10 - j*5)//2 + 1):  # перебор количества 2-рублевых монет
            ways += 1  # увеличиваем счетчик

print("Всего способов: ", ways)  # выводим результат

def get_ways(n, coins=[]):  # функция для получения всех способов
    if n == 0:  # если сумма нулевая, выводим список монет и купюр
        print(coins)
        return
    if n >= 10:  # выдаем 10-рублевую купюру
        get_ways(n-10, coins+[10])
    if n >= 5:  # выдаем 5-рублевую монету
        get_ways(n-5, coins+[5])
    if n >= 2:  # выдаем 2-рублевую монету
        get_ways(n-2, coins+[2])
    if n >= 1:  # выдаем 1-рублевую монету
        get_ways(n-1, coins+[1])

n = int(input("Введите сумму: "))  # вводим сумму
get_ways(n)  # вызываем функцию для получения всех способов
