n = int(input("Введите количество чисел: "))  # ввод количества чисел
numbers = []  # создание пустого списка для чисел
for i in range(n):
    x = int(input(f"Введите число {i+1}: "))
    numbers.append(x)  # добавление числа в список
b = int(input("Введите число b: "))  # ввод числа b
sum_numbers = sum(numbers)  # нахождение суммы чисел в списке
if sum_numbers % b == 0:  # проверка кратности
    print("Сумма чисел кратна числу b")
else:
    print("Сумма чисел не кратна числу b")
