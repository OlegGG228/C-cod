number = input("Введите четырехзначное число: ")
if "2" in number or "7" in number:
    print("Число содержит цифры 2 или 7")
else:
    print("Число не содержит цифры 2 или 7")

if "3" in number or "6" in number or "9" in number:
    print("Число содержит цифры 3, 6 или 9")
else:
    print("Число не содержит цифры 3, 6 или 9")
