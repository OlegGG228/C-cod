# Функция для нахождения наибольшего общего делителя двух чисел
def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)

# Основная функция для нахождения всех простых несократимых дробей
def main():
    fractions = [] # Список для хранения результата

    # Перебираем все возможные знаменатели
    for den in range(2, 8):
        # Перебираем все возможные числители для данного знаменателя
        for num in range(1, den):
            # Находим наибольший общий делитель числителя и знаменателя
            div = gcd(num, den)
            # Если НОД равен 1, то дробь несократимая
            if div == 1:
                fractions.append("{}/{}".format(num, den)) # Добавляем дробь в список

    # Выводим результат
    print("Простые несократимые дроби с знаменателями до 7:")
    for fraction in fractions:
        print(fraction)

if __name__ == '__main__':
    main()
