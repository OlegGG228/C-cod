a = float(input("Введите длину стороны a: "))
b = float(input("Введите длину стороны b: "))
c = float(input("Введите длину стороны c: "))
d = float(input("Введите длину стороны d: "))

if (a <= c and b <= d) or (a <= d and b <= c):
    print("Прямоугольник со сторонами", a, "и", b, "может уместиться внутри прямоугольника со сторонами", c, "и", d)
else:
    print("Прямоугольник со сторонами", a, "и", b, "нельзя уместить внутри прямоугольника со сторонами", c, "и", d)
