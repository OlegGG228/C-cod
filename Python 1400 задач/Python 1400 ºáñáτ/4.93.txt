def day_of_year(k, d):
    # определяем, каким днем недели было 1 января
    jan_1 = (d - 1) % 7 # остаток от деления на 7
    # определяем, каким днем недели будет k-й день года
    day_of_week = (jan_1 + k - 1) % 7
    if day_of_week == 5:
        return "суббота"
    elif day_of_week == 6:
        return "воскресенье"
    else:
        return "рабочий день"

# пример использования функции
k = 100 # номер дня в году
d = 1 # день недели, на который приходится 1 января
day = day_of_year(k, d)
print("День", k, "года -", day)
