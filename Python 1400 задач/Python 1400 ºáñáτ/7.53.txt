# вводим данные
p = int(input("Введите натуральное число p: "))
nums = []
for i in range(p):
    num = int(input("Введите число t" + str(i+1) + ": "))
    nums.append(num)

# вычисляем сумму чисел т., которые меньше 25,5
sum1 = 0
for num in nums:
    if num < 25.5:
        sum1 += num

# проверяем условие а)
if sum1 <= 50:
    print("Сумма чисел т., которые меньше 25,5, не превышает 50")
else:
    print("Сумма чисел т., которые меньше 25,5, превышает 50")

# вычисляем сумму чисел т., которые не превышают 20
sum2 = 0
for num in nums:
    if num <= 20:
        sum2 += num

# проверяем условие б)
if sum2 % 3 == 0:
    print("Сумма чисел т., которые не превышают 20, кратна трем")
else:
    print("Сумма чисел т., которые не превышают 20, не кратна трем")
