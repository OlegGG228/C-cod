r1 = 10.0 # сопротивление первого участка
r2 = 20.0 # сопротивление второго участка
v1 = 5.0 # напряжение на первом участке
v2 = 10.0 # напряжение на втором участке

# вычисляем ток, используя закон Ома
i1 = v1 / r1 # ток в первом участке
i2 = v2 / r2 # ток во втором участке

# сравниваем токи и выводим результат
if i1 < i2:
  print("Меньший ток протекает через первый участок.")
else:
  print("Меньший ток протекает через второй участок.")
