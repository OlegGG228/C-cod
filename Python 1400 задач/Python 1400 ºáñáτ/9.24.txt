a = 50 # начало интервала
b = 100 # конец интервала

max_div_num = 0 # максимальное количество делителей
max_dividers = [] # числа с максимальным количеством делителей

# перебираем числа в интервале от a до b
for num in range(a, b+1):
    dividers = [] # список делителей числа num
    for i in range(1, num+1):
        if num % i == 0:
            dividers.append(i) # добавляем делитель
    div_count = len(dividers) # количество делителей
    # если количество делителей больше текущего максимума,
    # то обновляем максимум и список чисел с максимальным количеством делителей
    if div_count > max_div_num:
        max_div_num = div_count
        max_dividers = [num]
    # если количество делителей равно текущему максимуму,
    # то добавляем число в список чисел с максимальным количеством делителей
    elif div_count == max_div_num:
        max_dividers.append(num)

max_dividers.sort() # сортируем список чисел с максимальным количеством делителей

# выводим результат
print("Число(-а) с максимальным количеством делителей ({}) на интервале от {} до {}: {}".format(max_div_num, a, b, max_dividers))
print("Максимальное из них: {}".format(max_dividers[-1]))
print("Минимальное из них: {}".format(max_dividers[0]))
