num_homes = int(input("Введите количество домов: "))
odd_sum = 0 # сумма жителей в домах с нечетными номерами
even_sum = 0 # сумма жителей в домах с четными номерами

for i in range(1, num_homes+1):
    residents = int(input("Введите количество жителей в доме №{}: ".format(i)))
    if i%2 == 0: # проверяем, четный ли номер дома
        even_sum += residents # если четный, добавляем жителей в сумму четных домов
    else:
        odd_sum += residents # если нечетный, добавляем жителей в сумму нечетных домов

if odd_sum > even_sum:
    print("Больше жителей проживает на стороне с нечетными домами")
elif even_sum > odd_sum:
    print("Больше жителей проживает на стороне с четными домами")
else:
    print("Одинаковое количество жителей на обеих сторонах улицы")
