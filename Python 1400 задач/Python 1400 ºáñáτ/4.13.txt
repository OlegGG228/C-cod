# объемы тел
v1 = 10.5 # в кубических метрах
v2 = 7.2 # в кубических метрах

# массы тел
m1 = 1500 # в килограммах
m2 = 1200 # в килограммах

# вычисление плотностей
rho1 = m1 / v1
rho2 = m2 / v2

# сравнение плотностей
if rho1 > rho2:
    print("Материал первого тела имеет большую плотность")
elif rho2 > rho1:
    print("Материал второго тела имеет большую плотность")
else:
    print("Плотности материалов равны")
