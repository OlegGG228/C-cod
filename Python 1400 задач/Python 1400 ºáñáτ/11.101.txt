import statistics

# список с количеством осадков за 15 лет (в мм)
osadki = [571.2, 451.3, 503.2, 405.7, 340.5, 437.2, 350.1, 399.5, 279.5, 241.1, 295.3, 367.6, 315.4, 418.5, 478.8]

# вычисляем среднее количество осадков
mean = statistics.mean(osadki)
print("Среднее количество осадков:", mean)

# вычисляем отклонение от среднего для каждого года
for i, os in enumerate(osadki):
    deviation = os - mean
    print("Отклонение от среднего в {} году: {}".format(i+1, deviation))
