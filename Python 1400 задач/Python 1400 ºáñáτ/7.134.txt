x = []

for i in range(15):
    x.append(float(input("Введите вещественное число: ")))

max_x = x[0] # предположим, что первый элемент - максимальный
min_x = x[0] # предположим, что первый элемент - минимальный

for i in range(1, 15):
    if x[i] > max_x: # если текущий элемент больше максимального
        max_x = x[i] # то текущий элемент становится максимальным
    if x[i] < min_x: # если текущий элемент меньше минимального
        min_x = x[i] # то текущий элемент становится минимальным

print("Максимальное вещественное число:", max_x)
print("Минимальное вещественное число:", min_x)
print("Максимальное и минимальное вещественные числа:", max_x, ",", min_x)
