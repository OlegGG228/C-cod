a = int(input("Введите первое число: "))
b = int(input("Введите второе число: "))
c = int(input("Введите третье число: "))

# самое большое
if a > b and a > c:
    print("Первое число является самым большим")
elif b > a and b > c:
    print("Второе число является самым большим")
else:
    print("Третье число является самым большим")

# самое маленькое
if a < b and a < c:
    print("Первое число является самым маленьким")
elif b < a and b < c:
    print("Второе число является самым маленьким")
else:
    print("Третье число является самым маленьким")

# среднее число
if (a > b and a < c) or (a < b and a > c):
    print("Первое число является средним")
elif (b > a and b < c) or (b < a and b > c):
    print("Второе число является средним")
else:
    print("Третье число является средним")
