# цены каждого предмета в первом наборе
set1_prices = [10, 20, 30, 40, 50, 60, 70, 80]

# цены каждого предмета во втором наборе
set2_prices = [15, 25, 35, 45, 55, 65, 75, 85]

# вычисляем общую сумму цен каждого набора
set1_total = sum(set1_prices)
set2_total = sum(set2_prices)

# сравниваем суммы цен для определения более дешевого набора
if set1_total < set2_total:
    print("Первый набор дешевле")
elif set2_total < set1_total:
    print("Второй набор дешевле")
else:
    print("Наборы стоят одинаково")
